Rank of a Matrix

The rank of a matrix is defined as (a) the maximum number of linearly independent column vectors in the matrix or (b) the maximum number of linearly independent row vectors in the matrix. Both definitions are equivalent.

Rank of a Matrix 
            const double EPS = 1E-9;

            int compute_rank(vector> A) {
                int n = A.size();
                int m = A[0].size();
            
                int rank = 0;
                vector row_selected(n, false);
                for (int i = 0; i < m; ++i) {
                    int j;
                    for (j = 0; j < n; ++j) {
                        if (!row_selected[j] && abs(A[j][i]) > EPS)
                            break;
                    }
            
                    if (j != n) {
                        ++rank;
                        row_selected[j] = true;
                        for (int p = i + 1; p < m; ++p)
                            A[j][p] /= A[j][i];
                        for (int k = 0; k < n; ++k) {
                            if (k != j && abs(A[k][i]) > EPS) {
                                for (int p = i + 1; p < m; ++p)
                                    A[k][p] -= A[j][p] * A[k][i];
                            }
                        }
                    }
                }
                return rank;
            }
